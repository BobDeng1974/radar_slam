329  void low_var_resamp( std::vector<double> &weights,
330  std::vector<Eigen::Affine3d> &particles )
331  {
332  // Create a cumulative sum of the particle weights:
333  std::vector<double> weights_cumsum = weights;
334  std::partial_sum( weights_cumsum.begin(), weights_cumsum.end(), weights_cumsum.begin() );
335  double W = weights_cumsum.back();
336 
337  // Draw a single sample from uniform(0,W):
338  dist_weights_ = std::uniform_real_distribution<double>( 0.0, W );
339  double r = dist_weights_( dist_gen_ );
340 
341  // Resample the particles:
342  std::vector<Eigen::Affine3d> particles_old = particles;
343  int p_ind = ( std::lower_bound( weights_cumsum.begin(), weights_cumsum.end(), r )
344  - weights_cumsum.begin() );
345 
346  particles.at( 0 ) = particles_old.at( p_ind );
347  for( int i = 1; i < num_particles; ++i )
348  {
349  p_ind = ( std::lower_bound( weights_cumsum.begin(), weights_cumsum.end(),
350  fmod( r + ( i * ( W / num_particles ) ), W ) )
351  - weights_cumsum.begin() );
352  particles.at( i ) = particles_old.at( p_ind );
353  }
354 
355  // Renormalize the weights:
356  double sum = std::accumulate( weights.begin(), weights.end(), 0.0 );
357  std::transform( weights.begin(), weights.end(), weights.begin(),
358  std::bind2nd( std::divides<double>(), sum ) );
359  }
